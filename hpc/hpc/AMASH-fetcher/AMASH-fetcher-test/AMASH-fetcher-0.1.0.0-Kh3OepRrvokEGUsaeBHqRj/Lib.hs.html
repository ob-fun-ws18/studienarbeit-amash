   22 -- | Fetches and prints the metadata for a given vendor id (e.g. "1210714")
   23 fetchVendorMetaData :: String -> IO ()
   24 fetchVendorMetaData vendorId = do
   25     let uri = URIs.vendor vendorId
   26         getJSON = simpleHttp uri
   27 
   28     e <- (eitherDecode <$> getJSON) :: IO (Either String Vendor.Vendor)
   29 
   30     case e of
   31         Left err -> putStrLn err
   32         Right vendorInfo -> do
   33             putStrLn "\n------------------------------"
   34             putStrLn $ "Got VendorInfo for: " ++ vendorId
   35             print vendorInfo
   36             putStrLn "------------------------------"
   37 
   38 -- | Fetches and prints the metadata for a given plugin key (e.g. "de.scandio.confluence.plugins.pocketquery")
   39 fetchPluginMetaData :: String -> IO ()
   40 fetchPluginMetaData pluginKey = do
   41     let uri = URIs.app pluginKey
   42         getJSON = simpleHttp uri
   43 
   44     e <- (eitherDecode <$> getJSON) :: IO (Either String App.App)

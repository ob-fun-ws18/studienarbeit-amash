   47 instance (FromBSON a, ToBSON a, Typeable a, Show a, Eq a) => Val a where
   48     val         x = Doc $ toBSON x
   49     cast' (Doc x) = fromBSON x
   50     cast' _       = Nothing

   53 -- TO BSON
   54 
   55 class ToBSON a where
   56     toBSON :: a -> Document
   57 
   58     default toBSON :: (Generic a, GConstructorCount (Rep a), GToBSON (Rep a)) => a -> Document
   59     toBSON a = genericToBSON (constructorCount a) (from a)
   60 
   61 class GToBSON f where
   62     genericToBSON :: Int -> f a -> Document
   63 
   64 -- | Unit type -> Empty document
   65 instance GToBSON U1 where
   66     genericToBSON _ U1 = []
   67 
   68 -- | Sum of types
   69 instance (GToBSON a, GToBSON b) => GToBSON (a :*: b) where
   70     genericToBSON n (x :*: y) = genericToBSON n x ++ genericToBSON n y
   71 
   72 -- | Product of types
   73 instance (GToBSON a, GToBSON b) => GToBSON (a :+: b) where
   74     genericToBSON n (L1 x) = genericToBSON n x
   75     genericToBSON n (R1 x) = genericToBSON n x
   76 
   77 -- | Datatype information tag
   78 instance (GToBSON a) => GToBSON (D1 c a) where
   79     genericToBSON n (M1 x) = genericToBSON n x
   80 
   81 -- | Constructor tag
   82 instance (GToBSON a, Constructor c) => GToBSON (C1 c a) where
   83     genericToBSON 0 (M1 x) = genericToBSON 0 x
   84     genericToBSON 1 (M1 x) = genericToBSON 1 x
   85     genericToBSON n c@(M1 x) = genericToBSON n x ++ [ constructorLabel =: conName c ]
   86 
   87 -- | Selector tag
   88 instance (Val a, Selector s) => GToBSON (S1 s (K1 i a)) where
   89     genericToBSON _ s@(M1 (K1 x)) = [TS.pack (selName s) =: x]
   90 
   91 -- | ObjectKey special treatment
   92 instance (Selector s) => GToBSON (S1 s (K1 i ObjectKey)) where
   93     genericToBSON _ (M1 (K1 (ObjectKey (Just key)))) = [ keyLabel =: key ]
   94     genericToBSON                              _ _ = []
   95 
   96 -- | Constants
   97 instance (ToBSON a) => GToBSON (K1 i a) where
   98     genericToBSON _ (K1 x) = toBSON x

  103 -- FROM BSON
  104 
  105 class FromBSON a where
  106     fromBSON :: Document -> Maybe a
  107 
  108     default fromBSON :: (Generic a, GConstructorCount (Rep a), GFromBSON (Rep a)) => Document -> Maybe a
  109     fromBSON doc = fmap to (genericFromBSON (constructorCount (undefined :: a)) doc)
  110 
  111 class GFromBSON f where
  112     genericFromBSON :: Int -> Document -> Maybe (f a)
  113 
  114 instance GFromBSON U1 where
  115     genericFromBSON _ _ = Just U1
  116 
  117 instance (GFromBSON a, GFromBSON b) => GFromBSON (a :*: b) where
  118     genericFromBSON n doc = do
  119         x <- genericFromBSON n doc
  120         y <- genericFromBSON n doc
  121         return $ x :*: y
  122 
  123 instance (GFromBSON a, GFromBSON b) => GFromBSON (a :+: b) where
  124     genericFromBSON n doc = left `mplus` right
  125         where
  126           left  = L1 <$> genericFromBSON n doc
  127           right = R1 <$> genericFromBSON n doc
  128 
  129 instance (GFromBSON a, Constructor c) => GFromBSON (C1 c a) where
  130     genericFromBSON 0 doc = M1 <$> genericFromBSON 0 doc
  131     genericFromBSON 1 doc = M1 <$> genericFromBSON 0 doc
  132     genericFromBSON n doc = do
  133         cname <- BSON.lookup constructorLabel doc
  134         if cname == conName (undefined :: M1 C c a r)
  135            then M1 <$> genericFromBSON n doc
  136            else Nothing
  137 
  138 instance (GFromBSON a) => GFromBSON (M1 D c a) where
  139     genericFromBSON n doc = M1 <$> genericFromBSON n doc
  140 
  141 instance (Val a, Selector s) => GFromBSON (S1 s (K1 i a)) where
  142     genericFromBSON _ doc = M1 . K1 <$> BSON.lookup sname doc
  143         where
  144           sname = TS.pack . selName $ (undefined :: S1 s (K1 i a) r)
  145 
  146 -- | ObjectKey special treatment
  147 instance (Selector s) => GFromBSON (S1 s (K1 i ObjectKey)) where
  148     genericFromBSON _ doc = Just . M1 . K1 $ ObjectKey (BSON.lookup keyLabel doc)

  153 class GConstructorCount f where
  154     gconstructorCount :: f a -> Int
  155 
  156 instance GConstructorCount V1 where
  157     gconstructorCount _ = 0
  158 
  159 instance (GConstructorCount a) => GConstructorCount (D1 d a) where
  160     gconstructorCount (M1 x) = gconstructorCount x
  161 
  162 instance (Constructor c) => GConstructorCount (C1 c a) where
  163     gconstructorCount _ = 1
  164 
  165 instance (GConstructorCount a, GConstructorCount b) => GConstructorCount (a :+: b) where
  166     gconstructorCount (_ :: (a :+: b) r) = gconstructorCount (undefined :: a r) +
  167                                            gconstructorCount (undefined :: b r)
  168 
  169 constructorCount :: (Generic a, GConstructorCount (Rep a)) => a -> Int
  170 constructorCount x = gconstructorCount $ from x

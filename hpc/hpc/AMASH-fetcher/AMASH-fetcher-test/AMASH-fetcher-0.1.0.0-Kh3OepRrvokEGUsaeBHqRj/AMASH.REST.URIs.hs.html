    1 -- | Contains / Generates URIs for the Atlassian Marketplace REST API.
    2 -- | All of these URIs are for GET calls.
    3 module AMASH.REST.URIs (
    4         baseURI,
    5 
    6         app,
    7         appMetrics,
    8         appPricing,
    9         appRecommendations,
   10         appReviews,
   11 
   12         apps,
   13         appsPaged,
   14         buildRankingURI,
   15 
   16         vendor,
   17         vendorContacts,
   18         vendorAppsPaged,
   19         vendorArchivedAppsPaged
   20     ) where
   21 
   22 import AMASH.Constants
   23 
   24 -----------------------------------------------------------------------
   25 
   26 -- | The Atlassian Marketplace REST base URI.
   27 baseURI :: String 
   28 baseURI = "https://marketplace.atlassian.com/rest/2"
   29 
   30 -----------------------------------------------------------------------
   31 
   32 -- | Provides the Atlassian Marketplace REST base URI.
   33 -- | https://developer.atlassian.com/platform/marketplace/rest/#api-addons-addonKey-get
   34 app :: String  -> String 
   35 app ""       = error "Empty addon key not allowed for URI buildung!"
   36 app addonKey = baseURI ++ "/addons/" ++ addonKey
   37 
   38 {-
   39 -- | https://developer.atlassian.com/platform/marketplace/rest/#api-addons-addonKey-versions-get
   40 appVersions :: String  -> String 
   41 appVersions addonKey = app addonKey ++ "/versions"
   42 
   43 -- | https://developer.atlassian.com/platform/marketplace/rest/#api-addons-addonKey-versions-latest-get
   44 appVersionLatest :: String  -> String 
   45 appVersionLatest addonKey = appVersions addonKey ++ "/latest"
   46 -}
   47 
   48 -- | https://developer.atlassian.com/platform/marketplace/rest/#api-addons-addonKey-distribution-get
   49 appMetrics :: String  -> String
   50 appMetrics addonKey = app addonKey ++ "/distribution"
   51 
   52 -- | https://developer.atlassian.com/platform/marketplace/rest/#api-addons-addonKey-pricing-cloudOrServer-liveOrPending-get
   53 appPricing, appPricing' :: String  -> String  -> String 
   54 appPricing addonKey "cloud"      = appPricing' addonKey "cloud"
   55 appPricing addonKey "datacenter" = appPricing' addonKey "datacenter"
   56 appPricing addonKey "server"     = appPricing' addonKey "server"
   57 appPricing addonKey pricingType  = error $ "Illegal pricing type \"" ++ pricingType ++ "\"! Allowed are only: [\"cloud\", \"datacenter\", \"server\"]."
   58 
   59 appPricing' addonKey pricingType = app addonKey ++ "/pricing/" ++ pricingType ++ "/live"
   60 
   61 -- | https://developer.atlassian.com/platform/marketplace/rest/#api-addons-addonKey-recommendations-get
   62 -- | The other apps that appear in the "recommendations" panel below your app.
   63 appRecommendations :: String  -> String 
   64 appRecommendations addonKey = app addonKey ++ "/recommendations?limit=25"
   65 
   66 -- | https://developer.atlassian.com/platform/marketplace/rest/#api-addons-addonKey-reviews-get
   67 appReviews :: String  -> String 
   68 appReviews addonKey = app addonKey ++ "/reviews"
   69 
   70 -----------------------------------------------------------------------
   71 
   72 apps :: String
   73 apps = baseURI ++ "/addons"
   74 
   75 appsPaged :: Integer -> String
   76 appsPaged page = baseURI ++ "/addons?" ++ pageParams page
   77 
   78 buildRankingURI :: Application -> AppsListFilter -> Integer -> String
   79 buildRankingURI application appsListFilter page = apps
   80                                         ++ "?application=" ++ showApplication application
   81                                         ++ "&filter=" ++ showInKebab appsListFilter
   82                                         ++ "&" ++ pageParams page
   83 
   84 -----------------------------------------------------------------------
   85 
   86 -- | https://developer.atlassian.com/platform/marketplace/rest/#api-vendors-vendorId-get
   87 vendor :: String -> String
   88 vendor vendorId = baseURI ++ "/vendors/" ++ vendorId
   89 
   90 -- | https://developer.atlassian.com/platform/marketplace/rest/#api-vendors-vendorId-contacts-get
   91 vendorContacts :: String -> String
   92 vendorContacts vendorId = vendor vendorId ++ "/contacts"
   93 
   94 vendorApps :: String -> String
   95 vendorApps vendorId = baseURI ++ "/addons/vendor/" ++ vendorId
   96 
   97 vendorAppsPaged :: String -> Integer -> String
   98 vendorAppsPaged vendorId page = vendorApps vendorId ++ "?" ++ pageParams page
   99 
  100 vendorArchivedApps :: String -> String
  101 vendorArchivedApps vendorId = baseURI ++ "/addons/archived/vendor/" ++ vendorId
  102 
  103 vendorArchivedAppsPaged :: String -> Integer -> String
  104 vendorArchivedAppsPaged vendorId page = vendorArchivedApps vendorId ++ "?" ++ pageParams page
  105 
  106 -----------------------------------------------------------------------
  107 
  108 pageParams :: Integer -> String
  109 pageParams page = "limit=" ++ show resultsPerPage ++ "&offset=" ++ show (page * resultsPerPage)
